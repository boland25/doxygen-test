45  unsigned long m, p, r;

50 
51  std::unique_ptr<KEY> keyp = std::make_unique<KEY>(*contextp);
52  if constexpr (std::is_same_v<KEY, helib::SecKey>) {
53  keyp = std::make_unique<helib::SecKey>(
54  helib::SecKey::readFrom(keyFile, *contextp));

60  return {std::move(contextp), std::move(keyp)};
61 }

63 inline long estimateCtxtSize(const helib::Context& context, long offset)
64 {
65  // Return in bytes.
66 
67  // We assume that the size of each element in the DCRT is BINIO_64BIT
68 
69  // sizeof(BINIO_EYE_CTXT_BEGIN) = 4;
70  // BINIO_32BIT = 4
71  // sizeof(long) = BINIO_64BIT = 8
72  // xdouble = s * sizeof(long) = 2 * BINIO_64BIT = 16
73 
74  // We assume that primeSet after encryption is context.ctxtPrimes
75  // We assume we have exactly 2 parts after encryption
76  // We assume that the DCRT prime set is the same as the ctxt one
77 
78  long size = 0;
79 
80  // Header metadata
81  size += 24;
82 
83  // Begin eye-catcher
84  size += 4;
85 
86  // Begin Ctxt metadata
87  // 64 = header_size = ptxtSpace (long) + intFactor (long) + ptxtMag (xdouble)
88  // + ratFactor (xdouble) + noiseBound (xdouble)
89  size += 64;
90 
91  // primeSet.write(str);
92  // size of set (long) + each prime (long)
93  size += 8 + context.getCtxtPrimes().card() * 8;
94 
95  // Begin Ctxt content size
96  // write_raw_vector(str, parts);
97  // Size of the parts vector (long)
98  size += 8;
99 
100  long part_size = 0;
101  // Begin CtxtPart size
102 
103  // skHandle.write(str);
104  // powerOfS (long) + powerOfX (long) + secretKeyID (long)
105  part_size += 24;
106 
107  // Begin DCRT size computation
108 
109  // this->DoubleCRT::write(str);
110  // map.getIndexSet().write(str);
111  // size of set (long) + each prime (long)
112  part_size += 8 + context.getCtxtPrimes().card() * 8;
113 
114  // DCRT data write as write_ntl_vec_long(str, map[i]);
115  // For each prime in the ctxt modulus chain
116  // size of DCRT column (long) + size of each element (long) +
117  // size of all the slots (column in DCRT) (PhiM long elements)
118  long dcrt_size = (8 + 8 * context.getPhiM()) * context.getCtxtPrimes().card();
119 
120  part_size += dcrt_size;
121 
122  // End DCRT size
123  // End CtxtPart size
124 
125  size += 2 * part_size; // 2 * because we assumed 2 parts
126  // End Ctxt content size
127 
128  // End eye-catcher
129  size += 4;
130 
131  return size + offset;
132 }
